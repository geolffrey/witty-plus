00068         dbo::Transaction transaction(_dbSession);

00070         // (even if they enter a hash of all zeros if user is not found they still won't be able to log in .. check 'result' below \/
00071         StdSHAValue hashFromDB(20, 0); // This'll be the hash we'll compare on.
00072         if (userFromDB)
00073             hashFromDB = userFromDB->passwordHash();
00074         // Check the password
00075         StdSHAValue hashFromNet = sha1(password);
00076         // Always compare 20 bytes, no shortcuts allowed. Don't wanna be hit by a timing attack
00077         assert(hashFromDB.size() == SHALength);
00078         // If there's no user .. it'll always be a false result .. marked as volatile to prevent optimization, to prevent timing attacks
00079         volatile bool result = userFromDB; 
00080         // Always do this loop and compare *every* element to prevent statistical timing attacks
00081         for(size_t i=0; i < SHALength; ++i) {
00082             if (hashFromDB[i] != hashFromNet[i])
00083                 result = false; // Don't just return from here .. make all checks constant time to prevent timing attacks
00084         }
00085         _userCache = userFromDB; // Store the logged in user
00086         transaction.commit();
00087         if (!result)
00088             userFromDB = dbo::ptr<User>();
00089         return userFromDB;
